import { databasePlugin, rest, veloxApp } from '@veloxts/velox'
import { db } from './config/database.js';
import { collections } from './router.js';

const app = await veloxApp({ port: 3030 });

  rest(collections, { prefix: '/api' })
);

await app.start();

// Create router and collections from procedure definitions

  healthProcedures,
  userProcedures,
);

export type AppRouter = typeof router;

  prefix: '/api'
}))

procedures/users.ts
// Zero codegen, full type inference
import { defineProcedures, procedure } from '@veloxts/router'
import { z } from 'zod'

export const userProcedures = defineProcedures('users', {
  getUser: procedure()
    .input(z.object({ id: z.string() }))
    .query(async ({ input, ctx }) => {
      return ctx.db.user.findUnique({
        where: { id: input.id }
      })
    }),

  createUser: procedure()
    .input(userSchema)
    .mutation(async ({ input, ctx }) => {
      return ctx.db.user.create({ data: input })
    })
})

// Type-safe client - no framework needed
import { createClient } from '@veloxts/client'
import type { AppRouter } from './api/router'

const api = createClient<AppRouter>({
  baseUrl: 'http://localhost:3030'
})

// Full autocomplete and type inference
const user = await api.users.getUser({
  id: '123'
})
